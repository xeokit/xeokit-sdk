    import {WEBGL_INFO} from '../../webglInfo.js';

const DEFAULT_RADIUS = 0.9;
const DEFAULT_INTENSITY = 50;
const DEFAULT_BIAS = 0.1;
const DEFAULT_SCALE = 0.9;
const DEFAULT_CUTOFF = 0.3;

const SAO_SOURCE = `
    #extension GL_OES_standard_derivatives : require
    precision highp float;
    #define PI2 6.283185
    #define TAU (PI2 * 2.0)
    #define NUM_SAMPLES 16.0
    #define SAMPLE_WEIGHT (1.0 / NUM_SAMPLES)
    #define EPSILON 0.0001
    varying vec2 vUV;
    uniform float uSampleRadius;
    uniform float uIntensity;
    uniform float uContrast;
    uniform float uBias;
    uniform vec2 uTexelOffset;
    uniform vec3 uDepthParams;
    uniform vec3 uXYParams;
    uniform sampler2D uDepthBuffer;
    vec3 getPosition(float fragDepth, vec2 uv) {
        vec2 clipCoord = uv * 2.0 - 1.0;
        float zView = uDepthParams.x / (fragDepth * uDepthParams.y + uDepthParams.z);
        return vec3(clipCoord.x * zView * uXYParams.x * uXYParams.y,clipCoord.y * zView * uXYParams.y,zView);
    }
    vec2 packDepth(float viewZ) {
        float d = viewZ / -uDepthParams.z;
        float temp = floor(d * 256.0);
        return vec2(temp / 256.0,d * 256.0 - temp);
    }
    void main() {
        float fragDepth = texture2D(uDepthBuffer, vUV).x;
        if (fragDepth == 1.0) { discard; }
        vec3 position = getPosition(fragDepth, vUV);
        vec3 normal = normalize(cross(dFdx(position), dFdy(position)));
        float ssRadius = min((uXYParams.z * uSampleRadius) / (position.z + EPSILON), 32.0);
        if (ssRadius < 4.0) { discard; }
        float occlusion = 0.0;
        for (int i = 0; i < int(NUM_SAMPLES); ++i) {
            float a = (float(i) + 0.5) * SAMPLE_WEIGHT;
            float h = ssRadius * a;
            float theta = a * TAU;
            vec2 u = vec2(cos(theta), sin(theta));
            vec2 uv = vUV + h * u * uTexelOffset;
            fragDepth = texture2D(uDepthBuffer, uv).x;
            vec3 samplePosition = getPosition(fragDepth, uv);
            vec3 dir = samplePosition - position;
            float len2 = dot(dir, dir);
            float f = max(0.0, uSampleRadius * uSampleRadius - len2);
            occlusion += f * f * f * max(0.0, dot(dir, normal) - uBias) / (len2 + EPSILON);
        }
        occlusion = max(0.0, occlusion * 2.0 * uIntensity * SAMPLE_WEIGHT);
        vec2 odd = mod(gl_FragCoord.xy, 2.0);
        if (abs(dFdx(position.z)) < 0.02) {
            occlusion -= dFdx(occlusion) * (odd.x - 0.5);
        }
        if (abs(dFdy(position.z)) < 0.02) {
            occlusion -= dFdy(occlusion) * (odd.y - 0.5);
        }
        gl_FragColor.r = occlusion;
        gl_FragColor.gb = packDepth(position.z);
    }`;

const SAO_BLUR_SOURCE = `
    precision highp float;
    varying vec2 vUV;
    uniform sampler2D uColorBuffer;
    uniform vec2 uTexelOffset;
    float unpackDepth(vec2 p) {
        return p.x + p.y / 256.0;
    }
    void main() {
        float gaussian[5];
        gaussian[0] = 0.153170;
        gaussian[1] = 0.144893;
        gaussian[2] = 0.122649;
        gaussian[3] = 0.092902;
        gaussian[4] = 0.062970;
        vec3 centerValue = texture2D(uColorBuffer, vUV).rgb;
        float totalWeight = gaussian[0];
        float sum = centerValue.r * totalWeight;
        float centerDepth = unpackDepth(centerValue.gb);
        for (int r = 1; r <= 4; ++r) {
            vec2 uv = vUV + uTexelOffset * float(r) * 2.0;
            vec3 sampleValue = texture2D(uColorBuffer, uv).rgb;
            float sampleDepth = unpackDepth(sampleValue.gb);
            float weight = gaussian[r];
            weight *= max(0.0, 1.0 - 10.0 * abs(sampleDepth - centerDepth));
            sum += sampleValue.r * weight;
            totalWeight += weight;
        }
        for (int r = 1; r <= 4; ++r) {
            vec2 uv = vUV + uTexelOffset * -float(r) * 2.0;
            vec3 sampleValue = texture2D(uColorBuffer, uv).rgb;
            float sampleDepth = unpackDepth(sampleValue.gb);
            float weight = gaussian[r];
            weight *= max(0.0, 1.0 - 10.0 * abs(sampleDepth - centerDepth));
            sum += sampleValue.r * weight;
            totalWeight += weight;
        }
        gl_FragColor.r = sum / (totalWeight + 0.0001);
    }`;

const SAO_BLEND_SOURCE = `
    precision highp float;
    varying vec2 vUV;
    uniform sampler2D uColorBuffer;
    uniform sampler2D uOcclusionBuffer;
    uniform float uOcclusionScale;
    uniform float uOcclusionCutoff;
    void main() {,
        vec4 color = texture2D(uColorBuffer, vUV);
        float ambient = 1.0 - smoothstep(uOcclusionCutoff, 1.0, texture2D(uOcclusionBuffer, vUV).r) * uOcclusionScale;
        gl_FragColor = vec4(color.rgb * ambient, color.a);
    }`;

class SAO {

    constructor(postEffects) {

        this.postEffects = postEffects;
        this.scene = postEffects.scene;
        this.supported = WEBGL_INFO.depthTexturesSupported;
        this.fbInitialized = false;
        this.sampleRadius = DEFAULT_RADIUS;
        this.intensity = DEFAULT_INTENSITY;
        this.bias = DEFAULT_BIAS;
        this.scale = DEFAULT_SCALE;

        this.saoConfig = {
            type: "sao",
            id: "sao",
            uniforms: {
                uSampleRadius: 0.4,
                uIntensity: 8,
                uBias: 0.001,
                uTexelOffset: new Float32Array([1 / window.innerWidth, 1 / window.innerHeight]),
                uXYParams: new Float32Array(3),
                uDepthParams: new Float32Array(3)
            },
            textures: {
                uDepthBuffer: {
                    framebuffer: "__MAIN",
                    target: "depth"
                }
            },
            framebuffer: "__BDS_SAO"
        };

        this.blurHConfig = {
            type: "saoblur",
            id: "saohblur",
            uniforms: {
                uTexelOffset: new Float32Array([1 / window.innerWidth, 0]),
            },
            textures: {
                uColorBuffer: {
                    framebuffer: "__BDS_SAO",
                    target: "color",
                }
            },
            framebuffer: "---SET_ME---"
        };

        this.blurVConfig = {
            type: "saoblur",
            id: "saovblur",
            uniforms: {
                uTexelOffset: new Float32Array([0, 1 / window.innerHeight]),
            },
            textures: {
                uColorBuffer: {
                    framebuffer: "---SET_ME---",
                    target: "color",
                }
            },
            framebuffer: "__BDS_SAO"
        };

        this.blendConfig = {
            type: "saoblend",
            id: "saoblend",
            uniforms: {
                uOcclusionScale: DEFAULT_SCALE,
                uOcclusionCutoff: DEFAULT_CUTOFF
            },
            textures: {
                uColorBuffer: {
                    framebuffer: "---SET_ME---",
                    target: "color",
                },
                uOcclusionBuffer: {
                    framebuffer: "__BDS_SAO",
                    target: "color"
                }
            },
            framebuffer: "--SET_ME---"
        };

        this.postEffects.createPostProcess("sao", SAO_SOURCE);
        this.postEffects.createPostProcess("saoblur", SAO_BLUR_SOURCE);
        this.postEffects.createPostProcess("saoblend", SAO_BLEND_SOURCE);
    }

    init(scene) {

    }

    addStage(cfg) {

        if (!this.supported) {
            return;
        }

        if (!this.fbInitialized) {
            this.initializeFramebuffer();
        }

        const canvas = this.scene.canvas;
        const canvasElement = canvas.canvas;
        var projection = this.scene.getProjection();
        var aspect = projection.aspect;
        var fov = projection.fov * Math.PI / 180;
        var near = projection.near;
        var far = projection.far;
        var multiplier = -Math.tan(fov * 0.5);
        var scaleFactor = -canvasElement.height / (2.0 * Math.tan(fov * 0.5));

        var width = canvasElement.width;
        var height = canvasElement.height;

        var ioFramebuffers = ctx.getIOFramebuffers("opaque");

        var colorIn = ioFramebuffers.read;
        var colorOut = ioFramebuffers.write;

        // In __MAIN depth, out: __XEOKIT_SAO
        this.saoConfig.uniforms.uSampleRadius = this.sampleRadius;
        this.saoConfig.uniforms.uIntensity = this.intensity;
        this.saoConfig.uniforms.uBias = this.bias;
        this.saoConfig.uniforms.uTexelOffset[0] = 1 / width;
        this.saoConfig.uniforms.uTexelOffset[1] = 1 / height;
        this.saoConfig.uniforms.uXYParams[0] = aspect;
        this.saoConfig.uniforms.uXYParams[1] = multiplier;
        this.saoConfig.uniforms.uXYParams[2] = scaleFactor;
        this.saoConfig.uniforms.uDepthParams[0] = -near * far;
        this.saoConfig.uniforms.uDepthParams[1] = near - far;
        this.saoConfig.uniforms.uDepthParams[2] = far;

        // In __XEOKIT_SAO, out: colorOut
        this.blurHConfig.uniforms.uTexelOffset[0] = 1 / width;
        this.blurHConfig.uniforms.uTexelOffset[1] = 0;
        this.blurHConfig.framebuffer = colorOut;

        // In colorOut, out: __XEOKIT_SAO
        this.blurVConfig.uniforms.uTexelOffset[0] = 0;
        this.blurVConfig.uniforms.uTexelOffset[1] = 1 / height;
        this.blurVConfig.textures.uColorBuffer.framebuffer = colorOut;

        // In colorIn, __XEOKIT_SAO, out: colorOut
        this.blendConfig.uniforms.uOcclusionScale = this.scale;
        this.blendConfig.textures.uColorBuffer.framebuffer = colorIn;
        this.blendConfig.framebuffer = colorOut;

        ctx.opaquePipeline.push(this.saoConfig);
        ctx.opaquePipeline.push(this.blurHConfig);
        ctx.opaquePipeline.push(this.blurVConfig);
        ctx.opaquePipeline.push(this.blendConfig);
    }

    initializeFramebuffer(scene) {
        this.scene.createFramebuffer("__XEOKIT_SAO", {
            depthAttachment: false
        });
    }

    setSampleRadius(r) {
        this.sampleRadius = this.saoConfig.uniforms.uSampleRadius = r;
        var saoStage = this.postEffects.getPostProcessStage("sao");
        if (saoStage) {
            saoStage.setUniform("uSampleRadius", this.sampleRadius);
        }
    }

    setIntensity(i) {
        this.intensity = this.saoConfig.uniforms.uIntensity = i;
        var saoStage = this.postEffects.getPostProcessStage("sao");
        if (saoStage) {
            saoStage.setUniform("uIntensity", this.intensity);
        }
    }

    setBias(b) {
        this.bias = this.saoConfig.uniforms.uBias = b;
        var saoStage = this.postEffects.getPostProcessStage("sao");
        if (saoStage) {
            saoStage.setUniform("uBias", this.bias);
        }
    }

    setOcclusionScale(s) {
        this.scale = this.blendConfig.uniforms.uOcclusionScale = s;
        var blendStage = this.postEffects.getPostProcessStage("saoblend");
        if (blendStage) {
            blendStage.setUniform("uOcclusionScale", this.scale);
        }
    }

    setOcclusionCutoff(c) {
        this.cutoff = this.blendConfig.uniforms.uOcclusionCutoff = c;
        var blendStage = this.postEffects.getPostProcessStage("saoblend");
        if (blendStage) {
            blendStage.setUniform("uOcclusionCutoff", this.cutoff);
        }
    };

    configure(cfg) {
        cfg = cfg || {};
        cfg.sampleRadius = cfg.sampleRadius === undefined ? DEFAULT_RADIUS : cfg.sampleRadius;
        cfg.intensity = cfg.intensity === undefined ? DEFAULT_INTENSITY : cfg.intensity;
        cfg.bias = cfg.bias === undefined ? DEFAULT_BIAS : cfg.bias;
        cfg.scale = cfg.scale === undefined ? DEFAULT_SCALE : cfg.scale;
        cfg.cutoff = cfg.cutoff === undefined ? DEFAULT_CUTOFF : cfg.cutoff;

        this.setSampleRadius(cfg.sampleRadius);
        this.setIntensity(cfg.intensity);
        this.setBias(cfg.bias);
        this.setOcclusionScale(cfg.scale);
        this.setOcclusionCutoff(cfg.cutoff);
    }

    updateProjection() {
        var projection = this.scene.getProjection();
        var aspect = projection.aspect;
        var fov = projection.fov * Math.PI / 180;
        var near = projection.near;
        var far = projection.far;
        var multiplier = -Math.tan(fov * 0.5);
        var scaleFactor = -Human.renderer.canvas.height / (2.0 * Math.tan(fov * 0.5));

        this.saoConfig.uniforms.uXYParams[0] = aspect;
        this.saoConfig.uniforms.uXYParams[1] = multiplier;
        this.saoConfig.uniforms.uXYParams[2] = scaleFactor;
        this.saoConfig.uniforms.uDepthParams[0] = -near * far;
        this.saoConfig.uniforms.uDepthParams[1] = near - far;
        this.saoConfig.uniforms.uDepthParams[2] = far;


        var saoStage = this.postEffects.getPostProcessStage("sao");

        if (saoStage) {
            saoStage.setUniform("uXYParams", this.saoConfig.uniforms.uXYParams);
            saoStage.setUniform("uDepthParams", this.saoConfig.uniforms.uDepthParams);
        }
    }

    updateCanvas() {
        const canvasBoundary = this.scene.canvas.boundary;
        var width = canvasBoundary[2];
        var height = canvasBoundary[3];

        var projection = this.scene.getProjection();
        var aspect = projection.aspect;
        var fov = projection.fov * Math.PI / 180;
        var multiplier = -Math.tan(fov * 0.5);
        var scaleFactor = -Human.renderer.canvas.height / (2.0 * Math.tan(fov * 0.5));

        this.saoConfig.uniforms.uTexelOffset[0] = 1 / width;
        this.saoConfig.uniforms.uTexelOffset[1] = 1 / height;
        this.saoConfig.uniforms.uXYParams[0] = aspect;
        this.saoConfig.uniforms.uXYParams[1] = multiplier;
        this.saoConfig.uniforms.uXYParams[2] = scaleFactor;

        this.blurHConfig.uniforms.uTexelOffset[0] = 1 / width;

        this.blurVConfig.uniforms.uTexelOffset[1] = 1 / height;

        var saoStage = this.postEffects.getPostProcessStage("sao");
        var blurHStage = this.postEffects.getPostProcessStage("saohblur");
        var blurVStage = this.postEffects.getPostProcessStage("saovblur");
        if (saoStage && blurHStage && blurVStage) {
            saoStage.setUniform("uXYParams", this.saoConfig.uniforms.uXYParams);
            saoStage.setUniform("uTexelOffset", this.saoConfig.uniforms.uTexelOffset);
            blurHStage.setUniform("uTexelOffset", this.blurHConfig.uniforms.uTexelOffset);
            blurVStage.setUniform("uTexelOffset",  this.blurVConfig.uniforms.uTexelOffset);
        }
    }
}
